# Obsidian HTML Viewer Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a native Obsidian plugin (`obsidian-html-viewer`) that renders `.html` files in workspace tabs, resolves local relative assets (CSS, images, scripts) via `vault.getResourcePath`, sandboxes execution with CSP, and intercepts links to open `.md` notes and `.html` files in Obsidian.

**Architecture:** Modular service architecture consisting of an entry point (`main.ts`), an `ItemView` wrapper (`HtmlView.ts`), a pure DOM path resolution engine (`HtmlAssetResolver.ts`), and an IPC message handler (`IpcBridge.ts`).

**Tech Stack:** TypeScript, Obsidian API, esbuild, Jest & JSDOM (for unit testing pure DOM functions).

## Global Constraints
- Plugin ID: `obsidian-html-viewer`
- Minimum Obsidian App Version: 0.15.0
- Sandboxing: `sandbox="allow-scripts allow-forms allow-same-origin allow-modals"`
- Navigation: Intercept `.md` and `.html` links for Obsidian tab workspace navigation, open external `http(s)` in system browser.

Create `src/HtmlAssetResolver.ts`:

```typescript
export interface ResolveResult {
  transformedHtml: string;
  assetPaths: string[];
}

/**
 * Normalizes relative file paths against a base folder path in the vault.
 */
function normalizeVaultPath(baseFolder: string, relativePath: string): string {
  const parts = (baseFolder ? baseFolder.split('/') : []).concat(relativePath.split('/'));
  const stack: string[] = [];

  for (const part of parts) {
    if (!part || part === '.') continue;
    if (part === '..') {
      if (stack.length > 0) stack.pop();
    } else {
      stack.push(part);
    }
  }

  return stack.join('/');
}

const CSP_META_TAG = `<meta http-equiv="Content-Security-Policy" content="default-src 'self' app: data: blob: 'unsafe-inline' 'unsafe-eval'; script-src 'self' app: data: blob: 'unsafe-inline' 'unsafe-eval'; style-src 'self' app: data: blob: 'unsafe-inline';">`;

const IPC_SCRIPT = `
<script>
(function() {
  document.addEventListener('click', function(e) {
    let target = e.target;
    while (target && target.tagName !== 'A') {
      target = target.parentElement;
    }
    if (!target) return;
    const href = target.getAttribute('href');
    if (!href) return;

    if (href.startsWith('http://') || href.startsWith('https://') || href.startsWith('mailto:')) {
      e.preventDefault();
      window.parent.postMessage({ type: 'obsidian-open-external', url: href }, '*');
    } else if (!href.startsWith('#') && !href.startsWith('javascript:')) {
      e.preventDefault();
      window.parent.postMessage({ type: 'obsidian-navigate', linkText: href }, '*');
    }
  }, true);
})();
</script>
`;

export function resolveHtmlAssets(
  rawHtml: string,
  currentFileFolderPath: string,
  getResourcePathFn: (path: string) => string
): ResolveResult {
  const parser = new DOMParser();
  const doc = parser.parseFromString(rawHtml, 'text/html');
  const assetPaths: string[] = [];

  // 1. Rewrite relative asset attributes
  const elementsToResolve = [
    { selector: 'link[rel="stylesheet"]', attr: 'href' },
    { selector: 'script[src]', attr: 'src' },
    { selector: 'img[src]', attr: 'src' },
    { selector: 'video[src]', attr: 'src' },
    { selector: 'audio[src]', attr: 'src' },
    { selector: 'source[src]', attr: 'src' }
  ];

  for (const { selector, attr } of elementsToResolve) {
    const nodes = doc.querySelectorAll(selector);
    nodes.forEach(node => {
      const originalVal = node.getAttribute(attr);
      if (!originalVal || originalVal.startsWith('http://') || originalVal.startsWith('https://') || originalVal.startsWith('data:') || originalVal.startsWith('blob:')) {
        return;
      }
      const normalized = normalizeVaultPath(currentFileFolderPath, originalVal);
      assetPaths.push(normalized);
      node.setAttribute(attr, getResourcePathFn(normalized));
    });
  }

  // 2. Ensure <head> exists and inject CSP + IPC Script
  let head = doc.querySelector('head');
  if (!head) {
    head = doc.createElement('head');
    if (doc.documentElement) {
      doc.documentElement.insertBefore(head, doc.documentElement.firstChild);
    }
  }

  const cspContainer = doc.createElement('div');
  cspContainer.innerHTML = CSP_META_TAG;
  if (cspContainer.firstChild) {
    head.appendChild(cspContainer.firstChild);
  }

  const ipcContainer = doc.createElement('div');
  ipcContainer.innerHTML = IPC_SCRIPT;
  if (ipcContainer.firstChild) {
    head.appendChild(ipcContainer.firstChild);
  }

  return {
    transformedHtml: doc.documentElement ? doc.documentElement.outerHTML : rawHtml,
    assetPaths
  };
}
```

Create `src/IpcBridge.ts`:

```typescript
import { App } from 'obsidian';

export interface IpcNavigatePayload {
  type: 'obsidian-navigate';
  linkText: string;
}

export interface IpcOpenExternalPayload {
  type: 'obsidian-open-external';
  url: string;
}

export type IpcPayload = IpcNavigatePayload | IpcOpenExternalPayload;

export class IpcBridge {
  private app: App;
  private getSourcePath: () => string;
  private messageHandler: (event: MessageEvent) => void;

  constructor(app: App, getSourcePath: () => string) {
    this.app = app;
    this.getSourcePath = getSourcePath;

    this.messageHandler = (event: MessageEvent) => {
      const data = event.data as IpcPayload;
      if (!data || typeof data !== 'object') return;

      if (data.type === 'obsidian-navigate') {
        const sourcePath = this.getSourcePath();
        this.app.workspace.openLinkText(data.linkText, sourcePath, false);
      } else if (data.type === 'obsidian-open-external') {
        window.open(data.url, '_blank');
      }
    };
  }

  public attach(): void {
    window.addEventListener('message', this.messageHandler);
  }

  public detach(): void {
    window.removeEventListener('message', this.messageHandler);
  }
}
```

Create `src/HtmlView.ts`:

```typescript
import { ItemView, WorkspaceLeaf, TFile } from 'obsidian';
import { resolveHtmlAssets } from './HtmlAssetResolver';
import { IpcBridge } from './IpcBridge';

export const VIEW_TYPE_HTML = 'html-viewer-view';

export class HtmlView extends ItemView {
  private mode: 'preview' | 'source' = 'preview';
  private containerElDiv: HTMLDivElement;
  private iframeEl: HTMLIFrameElement;
  private sourceEl: HTMLDivElement;
  private ipcBridge: IpcBridge;
  private trackedDependencies: Set<string> = new Set();

  constructor(leaf: WorkspaceLeaf) {
    super(leaf);
    this.ipcBridge = new IpcBridge(this.app, () => this.file?.path || '');
  }

  public getViewType(): string {
    return VIEW_TYPE_HTML;
  }

  public getDisplayText(): string {
    return this.file ? this.file.basename : 'HTML Viewer';
  }

  public getIcon(): string {
    return 'code';
  }

  public async onOpen(): Promise<void> {
    this.ipcBridge.attach();

    // Add header toggle action
    this.addAction('code', 'Toggle Source / Preview', () => this.toggleMode());
    this.addAction('refresh-cw', 'Reload Preview', () => this.renderView());

    const container = this.containerEl.children[1] as HTMLElement;
    container.empty();
    container.addClass('html-viewer-container');

    this.containerElDiv = container.createDiv({ cls: 'html-viewer-container' });

    this.iframeEl = this.containerElDiv.createEl('iframe', { cls: 'html-viewer-iframe' });
    this.iframeEl.setAttribute('sandbox', 'allow-scripts allow-forms allow-same-origin allow-modals');

    this.sourceEl = this.containerElDiv.createDiv({ cls: 'html-viewer-source-container' });
    this.sourceEl.style.display = 'none';

    // Register vault modification listener for live reloading
    this.registerEvent(
      this.app.vault.on('modify', (file: TFile) => {
        if (!this.file) return;
        if (file.path === this.file.path || this.trackedDependencies.has(file.path)) {
          this.renderView();
        }
      })
    );
  }

  public async onClose(): Promise<void> {
    this.ipcBridge.detach();
  }

  public async onLoadFile(file: TFile): Promise<void> {
    await super.onLoadFile(file);
    await this.renderView();
  }

  private toggleMode(): void {
    this.mode = this.mode === 'preview' ? 'source' : 'preview';
    if (this.mode === 'preview') {
      this.iframeEl.style.display = 'block';
      this.sourceEl.style.display = 'none';
    } else {
      this.iframeEl.style.display = 'none';
      this.sourceEl.style.display = 'block';
    }
  }

  private async renderView(): Promise<void> {
    if (!this.file) return;

    try {
      const rawHtml = await this.app.vault.read(this.file);
      this.sourceEl.textContent = rawHtml;

      const folderPath = this.file.parent ? this.file.parent.path : '';
      const resolved = resolveHtmlAssets(
        rawHtml,
        folderPath,
        (path) => {
          const file = this.app.vault.getAbstractFileByPath(path);
          if (file instanceof TFile) {
            return this.app.vault.getResourcePath(file);
          }
          return path;
        }
      );

      this.trackedDependencies = new Set(resolved.assetPaths);
      this.iframeEl.srcdoc = resolved.transformedHtml;
    } catch (err) {
      console.error('Error rendering HTML view:', err);
    }
  }
}
```

```typescript
import { Plugin, TFile, WorkspaceLeaf } from 'obsidian';
import { HtmlView, VIEW_TYPE_HTML } from './HtmlView';

export default class HtmlViewerPlugin extends Plugin {
  async onload(): Promise<void> {
    console.log('Loading HTML Viewer Plugin');

    this.registerView(
      VIEW_TYPE_HTML,
      (leaf: WorkspaceLeaf) => new HtmlView(leaf)
    );

    this.registerExtensions(['html'], VIEW_TYPE_HTML);

    this.addCommand({
      id: 'open-html-preview',
      name: 'Open active HTML file in preview',
      checkCallback: (checking: boolean) => {
        const file = this.app.workspace.getActiveFile();
        if (file && file.extension === 'html') {
          if (!checking) {
            this.activateView(file);
          }
          return true;
        }
        return false;
      }
    });
  }

  async onunload(): Promise<void> {
    console.log('Unloading HTML Viewer Plugin');
  }

  private async activateView(file: TFile): Promise<void> {
    const leaf = this.app.workspace.getLeaf(true);
    await leaf.setViewState({
      type: VIEW_TYPE_HTML,
      active: true,
    });
    if (leaf.view instanceof HtmlView) {
      await leaf.view.onLoadFile(file);
    }
  }
}
```
